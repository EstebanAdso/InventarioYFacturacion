SET SQL_SAFE_UPDATES = 1;

DELIMITER //

CREATE TRIGGER ActualizarEstadoProducto
BEFORE UPDATE ON producto
FOR EACH ROW
BEGIN
    IF NEW.cantidad <= 0 THEN
        SET NEW.estado = 'inactivo';
    ELSE
        SET NEW.estado = 'activo';
    END IF;
END;

DELIMITER ;
